96 {

99  if(argc < 3 ||

104  (strcmp(argv[2], "stoppub") != 0)))
105  {

110  std::string writer_action = argv[2];
111 
112  try
113  {
115  dds::domain::DomainParticipant dp(org::opensplice::domain::default_id());
116 
121  dds::topic::qos::TopicQos topicQos

123  << dds::core::policy::Durability::Transient()
124  << dds::core::policy::Reliability::Reliable();
125 
127  dds::topic::Topic<LifecycleData::Msg> topic(dp, "LifecycleData_Msg", topicQos);
128 
130  std::string name = "Lifecycle example";
131  dds::pub::qos::PublisherQos pubQos

134  dds::pub::Publisher pub(dp, pubQos);
135 
141  dds::pub::qos::DataWriterQos dwqos = topic.qos();
142  dwqos << dds::core::policy::WriterDataLifecycle(autodispose_unregistered_instances);
143 
146  dds::pub::DataWriter<LifecycleData::Msg> dw(pub, topic, dwqos);

153  exampleSleepMilliseconds(500);

171  }
