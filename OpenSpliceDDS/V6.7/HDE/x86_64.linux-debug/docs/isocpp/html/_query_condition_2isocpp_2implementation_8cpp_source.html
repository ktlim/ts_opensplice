176 
178  std::cout << "=== [QueryConditionDataQuerySubscriber] Query : ticker = " << requested_ticker << std::endl;
179  std::vector<std::string> params;
180  params.push_back(requested_ticker);
181  dds::sub::Query query(dr, "ticker=%0", params);
182  dds::sub::status::DataState anyDataState;
183 
186  bool closed = false;
187  unsigned int count = 0;
188  unsigned int correct_quote_count = 0;
189  do
190  {
191  dds::sub::LoanedSamples<StockMarket::Stock> samples;
192  /* The below two approaches are equivalent so we can just alternate between them */
193  if (correct_quote_count % 2)
194  {
195  /* take type #2 */
196  dr >> dds::sub::content(query)
197  >> dds::sub::state(dds::sub::status::DataState::any())
198  >> dds::sub::take

200  }
201  else

203  /* take type #1 */
204  samples = dr.select().content(query).state(dds::sub::status::DataState::any()).take();
205  }
206  for (dds::sub::LoanedSamples<StockMarket::Stock>::const_iterator sample = samples.begin();
207  sample < samples.end();
208  ++sample)
209  {
210  if (sample->info().valid())
211  {
212  if(sample->data().price() == -1.0f)
213  {
214  closed = true;
215  break;
216  }
217  std::cout << "=== [QueryConditionDataSubscriber] receives stockQuote : ("
218  << sample->data().ticker() << ": " << sample->data().price()
219  << ") using 'take' type #" << (correct_quote_count % 2) + 1 << std::endl;
220 
221  /* Verify the value is as expected */
222  if (requested_ticker == sample->data().ticker())
223  ++correct_quote_count;
224  else
225  {
226  std::cerr << "=== [QueryConditionDataQuerySubscriber] Unexpected quote received for "
227  << sample->data().ticker() << std::endl;
228  result = 1;
229  }
230  }
231  }
232  exampleSleepMilliseconds(900);
233  ++count;
234  }
235  while (!closed && count < write_loop_count * 30);
236 
237  if (!closed)
238  {
239  std::cerr << "ERROR: Waited for 60 seconds but no sample received" << std::endl;
240  result = 1;
241  }
242  else
243  {
244  std::cout << "=== [QueryConditionDataQuerySubscriber] Market Closed" << std::endl;
245  if (correct_quote_count != (requested_ticker == "GE" || requested_ticker == "MSFT" ? write_loop_count : 0))
246  {
247  std::cerr << "=== [QueryConditionDataQuerySubscriber] Incorrect quote count." << std::endl;
248  result = 1;
249  }
250  }
251  }

253  {
