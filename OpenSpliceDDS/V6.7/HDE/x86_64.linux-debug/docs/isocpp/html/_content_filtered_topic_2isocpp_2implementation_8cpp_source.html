161 
163  dds::sub::qos::DataReaderQos drqos = topic.qos();
164 
167  std::stringstream ss;
168  ss << "ticker = '" << requested_ticker << "'";
169  dds::topic::Filter filter(ss.str());
170  dds::topic::ContentFilteredTopic<StockMarket::Stock> cftopic(topic, "CFStockTrackerExclusive", filter);
171 
173  dds::sub::DataReader<StockMarket::Stock> dr(sub, cftopic, drqos);
174 
177  bool terminate_sample_received = false;
178  unsigned int correct_quote_count = 0;
179  for (unsigned int count = 0; count < write_loop_count * 10 && !terminate_sample_received; ++count)
180  {
181  dds::sub::LoanedSamples<StockMarket::Stock> samples = dr.take();
182  for (dds::sub::LoanedSamples<StockMarket::Stock>::const_iterator sample = samples.begin();
183  sample < samples.end();
184  ++sample)
185  {
186  if(sample->info().valid())
187  {
188  if (sample->data().price() == -1)
189  {
190  terminate_sample_received = true;
191  break;
192  }
193  std::cout << "=== [ContentFilteredTopicDataSubscriber] receives stockQuote : ("
194  << sample->data().ticker() << ", " << sample->data().price() << ")" << std::endl;
195 
196  if (requested_ticker == sample->data().ticker())
197  ++correct_quote_count;
198  else
199  {
200  std::cerr << "=== [ContentFilteredTopicDataSubscriber] Unexpected quote received for "
201  << sample->data().ticker() << std::endl;
202  result = 1;
203  }
204  }
205  }
206  exampleSleepMilliseconds(900);
207  }
208 
210  if (terminate_sample_received)
211  {
212  std::cout << "=== [ContentFilteredTopicDataSubscriber] Market Closed." << std::endl;
213  if (correct_quote_count != (requested_ticker == "GE" || requested_ticker == "MSFT" ? write_loop_count : 0))
214  {
215  std::cerr << "=== [ContentFilteredTopicDataSubscriber] Incorrect quote count." << std::endl;
216  result = 1;
217  }
218  }
219  else
220  {
221  std::cerr << "=== [ContentFilteredTopicDataSubscriber] Timed out waiting for close signal." << std::endl;
222  result = 1;
223  }
224  }
225  catch (const dds::core::Exception& e)
226  {
227  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
228  result = 1;
229  }
230  return result;
231 }
