409 
410  dds::core::cond::WaitSet::ConditionSeq conditions;
411  while(!terminated.trigger_value())
412  {
414  e.waitSet.wait(conditions);
415 
417  dds::sub::LoanedSamples<RoundTripModule::Sample> samples = e.reader.take();
418  for(dds::sub::LoanedSamples<RoundTripModule::Sample>::const_iterator sample = samples.begin();
419  !terminated.trigger_value() && sample < samples.end();
420  ++sample)
421  {
423  if(sample->info().state().instance_state() == dds::sub::status::InstanceState::not_alive_disposed())

427  break;
428  }
430  else if(sample->info().valid())
