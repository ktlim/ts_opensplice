142 int subscriber(int argc, char *argv[])
143 {
144  int result = 0;
145  try
146  {
150  if(argc < 2)
151  {
152  std::cerr << "*** [ContentFilteredTopicDataSubscriber] Subscription string not specified" << std::endl;
153  std::cerr << "*** usage : ContentFilteredTopicDataSubscriber <subscription_string>" << std::endl;

158 
161  dds::domain::DomainParticipant dp(org::opensplice::domain::default_id());
162  dds::topic::qos::TopicQos topicQos = dp.default_topic_qos()
163  << dds::core::policy::Durability::Transient()
164  << dds::core::policy::Reliability::Reliable();
165  dds::topic::Topic<StockMarket::Stock> topic(dp, "StockTrackerExclusive", topicQos);
166 
168  std::string name = "ContentFilteredTopic example";
169  dds::sub::qos::SubscriberQos subQos
170  = dp.default_subscriber_qos()
171  << dds::core::policy::Partition(name);
172  dds::sub::Subscriber sub(dp, subQos);
173 
175  dds::sub::qos::DataReaderQos drqos = topic.qos();
176 
179  std::stringstream ss;
180  ss << "ticker = '" << requested_ticker << "'";
181  dds::topic::Filter filter(ss.str());
182  dds::topic::ContentFilteredTopic<StockMarket::Stock> cftopic(topic, "CFStockTrackerExclusive", filter);
183 
185  dds::sub::DataReader<StockMarket::Stock> dr(sub, cftopic, drqos);
186 
189  bool terminate_sample_received = false;
190  unsigned int correct_quote_count = 0;
191  for (unsigned int count = 0; count < write_loop_count * 10 && !terminate_sample_received; ++count)
192  {
193  dds::sub::LoanedSamples<StockMarket::Stock> samples = dr.take();
194  for (dds::sub::LoanedSamples<StockMarket::Stock>::const_iterator sample = samples.begin();
195  sample < samples.end();
196  ++sample)
197  {
198  if(sample->info().valid())
199  {
200  if (sample->data().price() == -1)
201  {
202  terminate_sample_received = true;
203  break;
204  }
205  std::cout << "=== [ContentFilteredTopicDataSubscriber] receives stockQuote : ("
206  << sample->data().ticker() << ", " << sample->data().price() << ")" << std::endl;
207 
208  if (requested_ticker == sample->data().ticker())
209  ++correct_quote_count;
210  else
211  {
212  std::cerr << "=== [ContentFilteredTopicDataSubscriber] Unexpected quote received for "
213  << sample->data().ticker() << std::endl;
214  result = 1;
215  }
216  }
217  }
218  exampleSleepMilliseconds(900);
219  }
220 
222  if (terminate_sample_received)
223  {
224  std::cout << "=== [ContentFilteredTopicDataSubscriber] Market Closed." << std::endl;
225  if (correct_quote_count != (requested_ticker == "GE" || requested_ticker == "MSFT" ? write_loop_count : 0))
226  {
227  std::cerr << "=== [ContentFilteredTopicDataSubscriber] Incorrect quote count." << std::endl;
228  result = 1;
229  }
230  }
231  else
232  {
233  std::cerr << "=== [ContentFilteredTopicDataSubscriber] Timed out waiting for close signal." << std::endl;
234  result = 1;
235  }
236  }
237  catch (const dds::core::Exception& e)
238  {
239  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
240  result = 1;
241  }
242  return result;
243 }
