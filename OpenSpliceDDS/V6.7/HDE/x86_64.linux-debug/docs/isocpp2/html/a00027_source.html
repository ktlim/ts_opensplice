66  msgInstance.userID() = 1;
67  msgInstance.message() = "Lifecycle_1";

70  else if(writer_action == "unregister")

72  msgInstance.userID() = 2;
73  msgInstance.message() = "Lifecycle_2";

80  msgInstance.writerStates() = "SAMPLE_SENT -> DATAWRITER_DELETED";

85  std::cout << " Message : \"" << msgInstance.message() << "\"" << std::endl;
86  std::cout << " writerStates : \"" << msgInstance.writerStates() << "\"" << std::endl;

106 int publisher(int argc, char *argv[])
107 {
108  int result = 0;
109 
110  if(argc < 3 ||
111  ((strcmp(argv[1], "false") != 0) &&
112  (strcmp(argv[1], "true") != 0) &&
113  (strcmp(argv[2], "dispose") != 0) &&
114  (strcmp(argv[2], "unregister") != 0) &&
115  (strcmp(argv[2], "stoppub") != 0)))
116  {
117  usagePub();
118  return -1;
119  }
120  bool autodispose_unregistered_instances = (strcmp(argv[1], "true") == 0);
121  std::string writer_action = argv[2];
122 
123  try
124  {
126  dds::domain::DomainParticipant dp(org::opensplice::domain::default_id());
127 
132  dds::topic::qos::TopicQos topicQos
133  = dp.default_topic_qos()
134  << dds::core::policy::Durability::Transient()
135  << dds::core::policy::Reliability::Reliable();
136 
138  dds::topic::Topic<LifecycleData::Msg> topic(dp, "LifecycleData_Msg", topicQos);
139 
141  std::string name = "Lifecycle example";
142  dds::pub::qos::PublisherQos pubQos
143  = dp.default_publisher_qos()
144  << dds::core::policy::Partition(name);
145  dds::pub::Publisher pub(dp, pubQos);
146 
152  dds::pub::qos::DataWriterQos dwqos = topic.qos();
153  dwqos << dds::core::policy::WriterDataLifecycle(autodispose_unregistered_instances);
154 
157  dds::pub::DataWriter<LifecycleData::Msg> dw(pub, topic, dwqos);
158  dds::pub::DataWriter<LifecycleData::Msg> dwStopper(pub, topic, dwqos);
159  exampleSleepMilliseconds(500);
160 
161  writeSample(writer_action, dw);
162 
163  std::cout << "=== [Publisher] waiting 500ms to let the subscriber handle the previous write state ..." << std::endl;
164  exampleSleepMilliseconds(500);
165 
167  dw = dds::pub::DataWriter<LifecycleData::Msg>(dds::core::null);
168  exampleSleepMilliseconds(500);
169 
171  LifecycleData::Msg msgInstance(4, "Lifecycle_4", "STOPPING_SUBSCRIBER");
172  std::cout << "=== [Publisher] :" << std::endl;
173  std::cout << " userID : " << msgInstance.userID() << std::endl;
174  std::cout << " Message : \"" << msgInstance.message() << "\"" << std::endl;
175  std::cout << " writerStates : \"" << msgInstance.writerStates() << "\"" << std::endl;
176  dwStopper << msgInstance;
177 
178  /* A sleep ensures time is allowed for the sample to be written to the network.
179  If the example is running in *Single Process Mode* exiting immediately might
180  otherwise shutdown the domain services before this could occur */
181  exampleSleepMilliseconds(1000);
182  }
183  catch (const dds::core::Exception& e)
184  {
185  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
186  result = 1;
187  }
188  return result;
189 }

249 int subscriber(int argc, char *argv[])
250 {
251  int result = 0;
252  try
253  {
256  dds::domain::DomainParticipant dp(org::opensplice::domain::default_id());
257  dds::topic::qos::TopicQos topicQos = dp.default_topic_qos()
258  << dds::core::policy::Durability::Transient()
259  << dds::core::policy::Reliability::Reliable();
260  dds::topic::Topic<LifecycleData::Msg> topic(dp, "LifecycleData_Msg", topicQos);
261 
263  std::string name = "Lifecycle example";
264  dds::sub::qos::SubscriberQos subQos
265  = dp.default_subscriber_qos()
266  << dds::core::policy::Partition(name);
267  dds::sub::Subscriber sub(dp, subQos);
268 
270  dds::sub::qos::DataReaderQos drqos = topic.qos();
271 
273  dds::sub::DataReader<LifecycleData::Msg> dr(sub, topic, drqos);
274 
275  std::cout << "=== [Subscriber] Ready ..." << std::endl;
276 
279  bool closed = false;
280  int count = 0;
281  int maxCount = 100;
282  do
283  {
284  dds::sub::LoanedSamples<LifecycleData::Msg> samples = dr.read();
285  for (dds::sub::LoanedSamples<LifecycleData::Msg>::const_iterator sample = samples.begin();
286  sample < samples.end();
287  ++sample)
288  {
289  std::cout << "=== [Subscriber] message received :" << std::endl;
290  std::cout << " userID : " << sample->data().userID() << std::endl;
291  std::cout << " Message : \"" << sample->data().message() << "\"" << std::endl;
292  std::cout << " writerStates : \"" << sample->data().writerStates() << "\"" << std::endl;
293  std::cout << " valid_data : " << sample->info().valid() << std::endl;
294  std::cout << " sample_state:" << sampleStateToString(sample->info().state().sample_state());
295  std::cout << "-view_state:" << viewStateToString(sample->info().state().view_state());
296  std::cout << "-instance_state:" << instanceStateToString(sample->info().state().instance_state()) << std::endl;
297  closed = (sample->data().writerStates() == "STOPPING_SUBSCRIBER");
298  ++count;
299  }
300  exampleSleepMilliseconds(200);
301  }
302  while (!closed && count < maxCount);
303 
304  std::cout << "=== [Subscriber] stopping after " << count << " iterations - closed = " << closed << std::endl;
305  if (count == maxCount)
306  {
307  std::cout << "*** Error : max " << maxCount << " iterations reached" << std::endl;
308  result = 1;
309  }
310  }
311  catch (const dds::core::Exception& e)
312  {
313  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
314  result = 1;
315  }
316  return result;
317 }
