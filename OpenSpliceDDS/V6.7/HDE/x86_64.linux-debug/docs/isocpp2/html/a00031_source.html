421  dds::core::cond::WaitSet::ConditionSeq conditions;
422  while(!terminated.trigger_value())
423  {
425  e.waitSet.wait(conditions);
426 
428  dds::sub::LoanedSamples<RoundTripModule::DataType> samples = e.reader.take();
429  for(dds::sub::LoanedSamples<RoundTripModule::DataType>::const_iterator sample = samples.begin();
430  !terminated.trigger_value() && sample < samples.end();
431  ++sample)
432  {
434  if(sample->info().state().instance_state() == dds::sub::status::InstanceState::not_alive_disposed())
435  {
436  std::cout << "Received termination request. Terminating." << std::endl;
437  terminated.trigger_value(true);
438  break;
439  }
441  else if(sample->info().valid())
442  {

444  }

447  }
448  catch(const dds::core::Exception& e)
449  {

451  result = 1;
