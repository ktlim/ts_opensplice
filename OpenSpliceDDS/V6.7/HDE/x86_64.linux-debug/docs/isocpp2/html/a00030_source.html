151 {

153  try
154  {
155  // usage : QueryConditionDataQuerySubscriber <query_string>
156  std::string requested_ticker;
157  if (argc > 1)

160  }
161  else
162  {
163  std::cerr << "*** [QueryConditionDataQuerySubscriber] Query string not specified" << std::endl;
164  std::cerr << "*** usage : QueryConditionDataQuerySubscriber <query_string>" << std::endl;
165  return 1;
166  }
167 
170  dds::domain::DomainParticipant dp(org::opensplice::domain::default_id());
171  dds::topic::qos::TopicQos topicQos = dp.default_topic_qos()
172  << dds::core::policy::Durability::Transient()
173  << dds::core::policy::Reliability::Reliable();
174  dds::topic::Topic<StockMarket::Stock> topic(dp, "StockTrackerExclusive", topicQos);
175 
177  std::string name = "QueryCondition example";
178  dds::sub::qos::SubscriberQos subQos
179  = dp.default_subscriber_qos()
180  << dds::core::policy::Partition(name);
181  dds::sub::Subscriber sub(dp, subQos);
182 
184  dds::sub::qos::DataReaderQos drqos = topic.qos();
185 
187  dds::sub::DataReader<StockMarket::Stock> dr(sub, topic, drqos);
188 
190  std::cout << "=== [QueryConditionDataQuerySubscriber] Query : ticker = " << requested_ticker << std::endl;
191  std::vector<std::string> params;
192  params.push_back(requested_ticker);
193  dds::sub::Query query(dr, "ticker=%0", params);
194  dds::sub::status::DataState anyDataState;
195 
198  bool closed = false;
199  unsigned int count = 0;
200  unsigned int correct_quote_count = 0;
201  do
202  {
203  dds::sub::LoanedSamples<StockMarket::Stock> samples;
204  /* The below two approaches are equivalent so we can just alternate between them */
205  if (correct_quote_count % 2)
206  {
207  /* take type #2 */
208  dr >> dds::sub::content(query)
209  >> dds::sub::state(dds::sub::status::DataState::any())
210  >> dds::sub::take
211  >> samples;
212  }
213  else
214  {
215  /* take type #1 */
216  samples = dr.select().content(query).state(dds::sub::status::DataState::any()).take();
217  }
218  for (dds::sub::LoanedSamples<StockMarket::Stock>::const_iterator sample = samples.begin();
219  sample < samples.end();
220  ++sample)
221  {
222  if (sample->info().valid())
223  {
224  if(sample->data().price() == -1.0f)
225  {
226  closed = true;
227  break;
228  }
229  std::cout << "=== [QueryConditionDataSubscriber] receives stockQuote : ("
230  << sample->data().ticker() << ": " << sample->data().price()
231  << ") using 'take' type #" << (correct_quote_count % 2) + 1 << std::endl;
232 
233  /* Verify the value is as expected */
234  if (requested_ticker == sample->data().ticker())
235  ++correct_quote_count;
236  else
237  {
238  std::cerr << "=== [QueryConditionDataQuerySubscriber] Unexpected quote received for "
239  << sample->data().ticker() << std::endl;
240  result = 1;
241  }
242  }
243  }
244  exampleSleepMilliseconds(900);
245  ++count;
246  }
247  while (!closed && count < write_loop_count * 30);
248 
249  if (!closed)
250  {
251  std::cerr << "ERROR: Waited for 60 seconds but no sample received" << std::endl;
252  result = 1;
253  }
254  else
255  {
256  std::cout << "=== [QueryConditionDataQuerySubscriber] Market Closed" << std::endl;
257  if (correct_quote_count != (requested_ticker == "GE" || requested_ticker == "MSFT" ? write_loop_count : 0))
258  {
259  std::cerr << "=== [QueryConditionDataQuerySubscriber] Incorrect quote count." << std::endl;
260  result = 1;
261  }
262  }
263  }
264  catch (const dds::core::Exception& e)
265  {
266  std::cerr << "ERROR: Exception: " << e.what() << std::endl;
267  result = 1;
268  }
269  return result;
270 }
