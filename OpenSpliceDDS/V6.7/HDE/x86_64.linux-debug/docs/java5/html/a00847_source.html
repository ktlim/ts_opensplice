81  public static Set<Class<? extends Status>> convertMask(
82  OsplServiceEnvironment environment, int state) {
83  HashSet<Class<? extends Status>> statuses = new HashSet<Class<? extends Status>>();
84 
85  if (state == DDS.STATUS_MASK_ANY_V1_2.value) {
86  statuses.add(AllDataDisposedStatus.class);
87  statuses.add(DataAvailableStatus.class);
88  statuses.add(DataOnReadersStatus.class);
89  statuses.add(InconsistentTopicStatus.class);
90  statuses.add(LivelinessChangedStatus.class);
91  statuses.add(LivelinessLostStatus.class);
92  statuses.add(OfferedDeadlineMissedStatus.class);
93  statuses.add(OfferedIncompatibleQosStatus.class);
94  statuses.add(PublicationMatchedStatus.class);
95  statuses.add(RequestedDeadlineMissedStatus.class);
96  statuses.add(RequestedIncompatibleQosStatus.class);
97  statuses.add(SampleLostStatus.class);
98  statuses.add(SampleRejectedStatus.class);
99  statuses.add(SubscriptionMatchedStatus.class);
100  } else {
101  if (stateTest(state, DDS.ALL_DATA_DISPOSED_TOPIC_STATUS.value)) {
102  statuses.add(AllDataDisposedStatus.class);
103  }
104  if (stateTest(state, DDS.DATA_AVAILABLE_STATUS.value)) {
105  statuses.add(DataAvailableStatus.class);
106  }
107  if (stateTest(state, DDS.DATA_ON_READERS_STATUS.value)) {
108  statuses.add(DataOnReadersStatus.class);
109  }
110  if (stateTest(state, DDS.INCONSISTENT_TOPIC_STATUS.value)) {
111  statuses.add(InconsistentTopicStatus.class);
112  }
113  if (stateTest(state, DDS.LIVELINESS_CHANGED_STATUS.value)) {
114  statuses.add(LivelinessChangedStatus.class);
115  }
116  if (stateTest(state, DDS.LIVELINESS_LOST_STATUS.value)) {
117  statuses.add(LivelinessLostStatus.class);
118  }
119  if (stateTest(state, DDS.OFFERED_DEADLINE_MISSED_STATUS.value)) {
120  statuses.add(OfferedDeadlineMissedStatus.class);
121  }
122  if (stateTest(state, DDS.OFFERED_INCOMPATIBLE_QOS_STATUS.value)) {
123  statuses.add(OfferedIncompatibleQosStatus.class);
124  }
125  if (stateTest(state, DDS.PUBLICATION_MATCHED_STATUS.value)) {
126  statuses.add(PublicationMatchedStatus.class);
127  }
128  if (stateTest(state, DDS.REQUESTED_DEADLINE_MISSED_STATUS.value)) {
129  statuses.add(RequestedDeadlineMissedStatus.class);
130  }
131  if (stateTest(state, DDS.REQUESTED_INCOMPATIBLE_QOS_STATUS.value)) {
132  statuses.add(RequestedIncompatibleQosStatus.class);
133  }
134  if (stateTest(state, DDS.SAMPLE_LOST_STATUS.value)) {
135  statuses.add(SampleLostStatus.class);
136  }
137  if (stateTest(state, DDS.SAMPLE_REJECTED_STATUS.value)) {
138  statuses.add(SampleRejectedStatus.class);
139  }
140  if (stateTest(state, DDS.SUBSCRIPTION_MATCHED_STATUS.value)) {
141  statuses.add(SubscriptionMatchedStatus.class);
142  }
143  }
144  return statuses;
145  }

156  public static int convertMask(OsplServiceEnvironment env,
157  Collection<Class<? extends Status>> statuses) {
158  int mask;
159 
160  if (statuses == null) {
161  return DDS.STATUS_MASK_ANY_V1_2.value;
162  } else if (statuses.size() == 0) {
163  return DDS.STATUS_MASK_NONE.value;
164  } else if (statuses.size() == 1 && statuses.iterator().next() == null) {
165  return DDS.STATUS_MASK_ANY_V1_2.value;
166  }
167  mask = DDS.STATUS_MASK_NONE.value;
168 
169  for (Class<? extends Status> clz : statuses) {
170  if (clz == null) {
171  throw new IllegalArgumentExceptionImpl(env,
172  "Passed illegal <null> status.");
173  } else if (clz.equals(DataAvailableStatus.class)) {
174  mask |= DDS.DATA_AVAILABLE_STATUS.value;
175  } else if (clz.equals(InconsistentTopicStatus.class)) {
176  mask |= DDS.INCONSISTENT_TOPIC_STATUS.value;
177  } else if (clz.equals(OfferedDeadlineMissedStatus.class)) {
178  mask |= DDS.OFFERED_DEADLINE_MISSED_STATUS.value;
179  } else if (clz.equals(RequestedDeadlineMissedStatus.class)) {
180  mask |= DDS.REQUESTED_DEADLINE_MISSED_STATUS.value;
181  } else if (clz.equals(OfferedIncompatibleQosStatus.class)) {
182  mask |= DDS.OFFERED_INCOMPATIBLE_QOS_STATUS.value;
183  } else if (clz.equals(RequestedIncompatibleQosStatus.class)) {
184  mask |= DDS.REQUESTED_INCOMPATIBLE_QOS_STATUS.value;
185  } else if (clz.equals(SampleLostStatus.class)) {
186  mask |= DDS.SAMPLE_LOST_STATUS.value;
187  } else if (clz.equals(SampleRejectedStatus.class)) {
188  mask |= DDS.SAMPLE_REJECTED_STATUS.value;
189  } else if (clz.equals(DataOnReadersStatus.class)) {
190  mask |= DDS.DATA_ON_READERS_STATUS.value;
191  } else if (clz.equals(LivelinessLostStatus.class)) {
192  mask |= DDS.LIVELINESS_LOST_STATUS.value;
193  } else if (clz.equals(LivelinessChangedStatus.class)) {
194  mask |= DDS.LIVELINESS_CHANGED_STATUS.value;
195  } else if (clz.equals(PublicationMatchedStatus.class)) {
196  mask |= DDS.PUBLICATION_MATCHED_STATUS.value;
197  } else if (clz.equals(SubscriptionMatchedStatus.class)) {
198  mask |= DDS.SUBSCRIPTION_MATCHED_STATUS.value;
199  } else if (clz.equals(AllDataDisposedStatus.class)) {
200  mask |= DDS.ALL_DATA_DISPOSED_TOPIC_STATUS.value;
201  } else if (clz.equals(Status.class)) {
202  throw new IllegalArgumentExceptionImpl(env,
203  "Provided class does not extend from the org.omg.dds.core.status.Status class.");
204  } else {
205  throw new IllegalArgumentExceptionImpl(env,
206  "Found illegal Class<? extends Status>: "
207  + clz.getName());
208  }
209  }
210  return mask;
211  }
