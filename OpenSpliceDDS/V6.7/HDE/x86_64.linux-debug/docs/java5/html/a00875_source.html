281  public static DataReaderQosImpl convert(OsplServiceEnvironment env,
282  DDS.DataReaderQos oldQos) {
283  if (oldQos == null) {
284  throw new IllegalArgumentExceptionImpl(env,
285  "oldQos parameter is null.");
286  }
287 
288  DataReaderQosImpl qos = new DataReaderQosImpl(env);
289 
290  qos.put(Deadline.class, PolicyConverter.convert(env, oldQos.deadline));
291  qos.put(DestinationOrder.class,
292  PolicyConverter.convert(env, oldQos.destination_order));
293  qos.put(Durability.class,
294  PolicyConverter.convert(env, oldQos.durability));
295  qos.put(History.class, PolicyConverter.convert(env, oldQos.history));
296  qos.put(LatencyBudget.class,
297  PolicyConverter.convert(env, oldQos.latency_budget));
298  qos.put(Liveliness.class,
299  PolicyConverter.convert(env, oldQos.liveliness));
300  qos.put(Ownership.class, PolicyConverter.convert(env, oldQos.ownership));
301  qos.put(ReaderDataLifecycle.class,
302  PolicyConverter.convert(env, oldQos.reader_data_lifecycle));
303  qos.put(Reliability.class,
304  PolicyConverter.convert(env, oldQos.reliability));
305  qos.put(ResourceLimits.class,
306  PolicyConverter.convert(env, oldQos.resource_limits));
307 
308  qos.put(TimeBasedFilter.class,
309  PolicyConverter.convert(env, oldQos.time_based_filter));
310  qos.put(UserData.class, PolicyConverter.convert(env, oldQos.user_data));
311 
312  Share share = PolicyConverter.convert(env, oldQos.share);
313 
314  if (share != null) {
315  qos.put(Share.class, share);
316  }
317 
318  ReaderLifespan readerLifespan = PolicyConverter.convert(env,
319  oldQos.reader_lifespan);
320 
321  if (readerLifespan != null) {
322  qos.put(ReaderLifespan.class, readerLifespan);
323  }
324  SubscriptionKeys subscriptionKeys = PolicyConverter.convert(env,
325  oldQos.subscription_keys);
326 
327  if (subscriptionKeys != null) {
328  qos.put(SubscriptionKeys.class, subscriptionKeys);
329  }
330  return qos;
