78  public static Scheduling convert(OsplServiceEnvironment env,
79  DDS.SchedulingQosPolicy old) {
80  Scheduling ls = new SchedulingImpl(env);
81 
82  switch (old.scheduling_class.kind.value()) {
83  case DDS.SchedulingClassQosPolicyKind._SCHEDULE_DEFAULT:
84  ls = ls.withSchedulingClass(SchedulingClass.DEFAULT);
85  break;
86  case DDS.SchedulingClassQosPolicyKind._SCHEDULE_REALTIME:
87  ls = ls.withSchedulingClass(SchedulingClass.REALTIME);
88  break;
89  case DDS.SchedulingClassQosPolicyKind._SCHEDULE_TIMESHARING:
90  ls = ls.withSchedulingClass(SchedulingClass.TIMESHARING);
91  break;
92  default:
93  throw new DDSExceptionImpl(env,
94  "Failed to convert listenerSchedulingClass");
95  }
96  switch (old.scheduling_priority_kind.kind.value()) {
97  case DDS.SchedulingPriorityQosPolicyKind._PRIORITY_ABSOLUTE:
98  ls = ls.withKind(SchedulingKind.ABSOLUTE);
99  break;
100  case DDS.SchedulingPriorityQosPolicyKind._PRIORITY_RELATIVE:
101  ls = ls.withKind(SchedulingKind.RELATIVE);
102  break;
103  default:
104  throw new DDSExceptionImpl(env,
105  "Failed to convert listenerSchedulingKind");
106  }
107  return ls;
108  }
